这种在运行时，动态地将代码切入到类的指定方法、指定位置上的编程思想就是面向切面的编程

方式一: 使用原生spring api接口

在applicationContext.xml中配置

<!-- 注册bean -->
<bean id="log" class="com.chun.log">
<!-- 配置aop：导入aop约束 -->
<aop:config>
    <!-- execution(修饰符 返回值 包名.类名/接口名.方法名(参数列表))
        - 返回类型、方法名、参数是必须有的.
        - *表示任意值. 比如返回类型，方法名等.
        - 参数(..) 代表任意参数
    -->
    <!-- 切入点 -->
    <aop:pointcut id="pointcut" expression="execution(* com.chun.service.UserServiceImpl.*(..))"/>
    <!-- 执行环绕增加  把log这个类切入到这个切入点-->
    <aop:advisor advice-ref="log" pointcut-ref="pointcut"/>

</aop:config>
